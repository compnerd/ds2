97 // We use this weird array_sizeof implementation to get the number of elements
98 // in an array in two cases we care about:
99 // 1) simple static arrays (e.g.: int foo[10]);
100 // 2) more complex structures that have a static array member but use an
101 // overload of operator[] to access data, and apply some transforms to the
102 // index passed. This happens in CPUState functions.
103 //
104 // array_sizeof is enabled only if the argument is not a pointer (to avoid the
105 // classic sizeof(pointer) bug), and if the argument is a POD type (to avoid
106 // users doing array_sizeof(my_std_vector) and such. The idea is that if a type
107 // overloads operator[] and is a POD, the underlying storage is probably in the
108 // structure, and not held as a reference or a pointer.
109 template <typename T>
110 typename std::enable_if<
111  !std::is_pointer<T>::value && std::is_pod<T>::value,
112  size_t>::type static inline constexpr array_sizeof(T const &array) {
113  return sizeof(array) / (reinterpret_cast<uintptr_t>(&array[1]) -
114  reinterpret_cast<uintptr_t>(&array[0]));
115 }

125 // This thing allows classes with protected constructors to call
126 // make_protected_unique to construct a unique_ptr cleanly (which calls
127 // make_unique internally). Without this, classes with protected constructors
128 // cannot be make_unique'd.
129 template <typename T> struct make_unique_enabler {
130  struct make_unique_enabler_helper : public T {
131  template <typename... Args>
132  make_unique_enabler_helper(Args... args) : T(std::forward<Args>(args)...) {}
133  };
134 
135  template <typename... Args>
136  static std::unique_ptr<T> make_protected_unique(Args... args) {
137  return ds2::make_unique<make_unique_enabler_helper>(
138  std::forward<Args>(args)...);
139  }
140 };
